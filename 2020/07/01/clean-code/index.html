// 原先代码
int length = role.length();
if (length == 1) {
    if (role.equalsIgnoreCase("O")) {
        ret.setRole("O");
        ret.setRoleName("ONE");
    } else if (role.equalsIgnoreCase("I")) {
        ret.setRole("I");
        ret.setRoleName("TWO");
    } else if (role.equalsIgnoreCase("P")) {
        ret.setRole("P");
        ret.setRoleName("THREE");
    } else {
        ret.setRole("G");

    }
}

第三层逻辑

else if (length == 3) {
    if (!role.contains("G")) {// iop
        ret.setRole("OIP");
        ret.setRoleName("ONE&TWO&THREE");
    } else if (!role.contains("P")) {// iog
        ret.setRole("IOG");
        ret.setRoleName("TWO&ONE&FOUR");
    }
    if (!role.contains("I")) {// opg
        ret.setRole("OPG");
        ret.setRoleName("ONE&THREE&FOUR");
    }
} else if (length == 4) {
    ret.setRole("OIPG");
    ret.setRoleName("ONE&TWO&THREE&FOUR");
} else {
    ret.setRole("");
    ret.setRoleName("");
    roleName.append("&");
}

可以看出原先代码的逻辑虽然很简单，但是嵌套的if else判断中很多重复逻辑

// 修改之后
char[] roles = role.toCharArray();
for(char cRole:roles){
    switch (cRole){
        case 'O':
            roleName.append("ONE");
            break;
        case 'I':
            roleName.append("TWO");
            break;
        case 'P':
            roleName.append("THREE");
            break;
        case 'G':
            roleName.append("FOUR");
            break;
        default:
            log("role为：" + cRole);
    }
}
